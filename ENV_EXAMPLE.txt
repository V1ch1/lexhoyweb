# Ejemplo de variables de entorno necesarias para el sistema de notificaciones
# Copia estas variables a tu archivo .env.local

# Supabase
NEXT_PUBLIC_SUPABASE_URL=tu_supabase_url
NEXT_PUBLIC_SUPABASE_ANON_KEY=tu_supabase_anon_key

# WordPress
NEXT_PUBLIC_WP_USER=tu_usuario_wordpress
NEXT_PUBLIC_WP_APP_PASSWORD=tu_app_password_wordpress

# Resend Email Service (NUEVO)
RESEND_API_KEY=re_tu_api_key_aqui
RESEND_FROM_EMAIL=notificaciones@lexhoy.com

# Base URL (para links en emails)
NEXT_PUBLIC_BASE_URL=http://localhost:3000

# - RESEND_API_KEY: Obtener de https://resend.com/api-keys
# - RESEND_FROM_EMAIL: Si no tienes dominio verificado, usa: onboarding@resend.dev
